def initialize(size, coeff)
  @size = size.floor
  @coeff = coeff
  @idx = 0
  @buf = Array.new(@size)
  @buf = @buf.map { |e| 0.0 }
end

def run(x)
  w = x - @buf[@idx] * @coeff;
  y = @buf[@idx] + w * @coeff;
  @buf[@idx] = w
  @idx = (@idx + 1) % @size;
  y
end
